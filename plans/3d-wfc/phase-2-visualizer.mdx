# Phase 2: Visualizer

> Part of [3D WFC](./README.mdx) · **5 story points**

## Goal

Create a Three.js voxel visualizer (`viz3d/`) that renders 3D WFC results with orbit camera, step-by-step animation, and z-axis slicing. After this phase:
- Browser at `http://localhost:3457` shows 3D voxel grid
- Step animation shows collapse in real-time
- Z-scrubber lets you slice into the volume
- Same SSE streaming pattern as 2D visualizer

## Skills to Reference
- `threejs-gotchas` — InstancedMesh for efficient voxel rendering, sRGB colors, dispose discipline
- `game-dev-gotchas` — zero-allocation render loops for smooth animation

## Knowledge to Reference
- `skills/threejs-optimization/skill.md` — instancing, draw call reduction, `renderer.info` monitoring
- `research/pixzels-pixel-art-perspective-tool-2026-06-20.md` — Three.js voxel rendering with custom BufferGeometry and exposed-face culling

### Sprint 2: Three.js Voxel Renderer
**Goal:** Render voxels using InstancedMesh.

- [ ] Task 2.1: Create `viz3d/index.html` with Three.js setup `(2pt)`
  ```html
  <!DOCTYPE html>
  <html>
  <head>
    <title>WFC 3D Visualizer</title>
    <script type="importmap">
    {
      "imports": {
        "three": "https://cdn.jsdelivr.net/npm/three@0.169/build/three.module.js",
        "three/addons/": "https://cdn.jsdelivr.net/npm/three@0.169/examples/jsm/"
      }
    }
    </script>
  </head>
  <body>
    <div id="controls">
      <button id="step">Step</button>
      <button id="run">Run</button>
      <button id="reset">Reset</button>
      <label>Z-Slice: <input type="range" id="z-slice" min="0" max="10" value="10"></label>
      <span id="status"></span>
    </div>
    <canvas id="canvas"></canvas>
    <script type="module">
      import * as THREE from 'three';
      import { OrbitControls } from 'three/addons/controls/OrbitControls.js';

      // Setup scene, camera, renderer
      const scene = new THREE.Scene();
      const camera = new THREE.PerspectiveCamera(60, window.innerWidth/window.innerHeight, 0.1, 100);
      camera.position.set(10, 10, 10);
      const renderer = new THREE.WebGLRenderer({ canvas: document.getElementById('canvas'), antialias: true });
      renderer.setSize(window.innerWidth, window.innerHeight);

      const controls = new OrbitControls(camera, renderer.domElement);

      // Lighting
      scene.add(new THREE.AmbientLight(0x404040));
      const dirLight = new THREE.DirectionalLight(0xffffff, 1);
      dirLight.position.set(5, 10, 5);
      scene.add(dirLight);

      // Voxel rendering via InstancedMesh
      // One InstancedMesh per tile type (for different colors)
      let voxelMeshes = [];
      const CUBE_GEO = new THREE.BoxGeometry(0.95, 0.95, 0.95);

      function setupVoxels(width, height, depth, colors) {
        // Dispose old meshes
        voxelMeshes.forEach(m => { scene.remove(m); m.geometry.dispose(); m.material.dispose(); });
        voxelMeshes = [];

        const count = width * height * depth;
        colors.forEach((color, tileIdx) => {
          const mat = new THREE.MeshStandardMaterial({ color });
          const mesh = new THREE.InstancedMesh(CUBE_GEO, mat, count);
          mesh.count = 0; // Start with no visible instances
          mesh.userData = { tileIdx, positions: [] };
          scene.add(mesh);
          voxelMeshes.push(mesh);
        });

        // Add unresolved mesh (translucent gray)
        const unresolvedMat = new THREE.MeshStandardMaterial({
          color: 0x888888,
          transparent: true,
          opacity: 0.3
        });
        const unresolvedMesh = new THREE.InstancedMesh(CUBE_GEO, unresolvedMat, count);
        unresolvedMesh.userData = { tileIdx: -1, positions: [] };
        scene.add(unresolvedMesh);
        voxelMeshes.push(unresolvedMesh);
      }

      function updateVoxels(result, width, height, depth, zSlice) {
        // Reset counts
        voxelMeshes.forEach(m => m.count = 0);

        const dummy = new THREE.Object3D();
        const counts = voxelMeshes.map(() => 0);

        for (let z = 0; z < depth && z <= zSlice; z++) {
          for (let y = 0; y < height; y++) {
            for (let x = 0; x < width; x++) {
              const i = x + y * width + z * width * height;
              const tile = result[i];
              const meshIdx = tile >= 0 ? tile : voxelMeshes.length - 1;
              const mesh = voxelMeshes[meshIdx];
              if (!mesh) continue;

              dummy.position.set(x - width/2, y - height/2, z - depth/2);
              dummy.updateMatrix();
              mesh.setMatrixAt(counts[meshIdx], dummy.matrix);
              counts[meshIdx]++;
            }
          }
        }

        voxelMeshes.forEach((m, idx) => {
          m.count = counts[idx];
          m.instanceMatrix.needsUpdate = true;
        });
      }

      // Animation loop
      function animate() {
        requestAnimationFrame(animate);
        controls.update();
        renderer.render(scene, camera);
      }
      animate();

      // ... SSE connection and UI handlers (see Task 2.2)
    </script>
  </body>
  </html>
  ```
  - DO NOT create individual cube meshes per voxel — use InstancedMesh
  - DO NOT forget to dispose old meshes on reset
