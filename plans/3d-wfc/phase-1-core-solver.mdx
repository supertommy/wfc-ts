- New `WFCSolver3D` can solve 3D voxel grids with 6-direction rules
- Tree-shaking works: importing only 2D doesn't bundle 3D topology

  ```typescript
  export interface Topology {
    readonly cellCount: number;
    readonly directionCount: number;
    /** Returns neighbor cell index for (cell, direction), or -1 if out of bounds */
    neighbor(cell: number, dir: number): number;
    /** Returns the opposite direction index */

    readonly directionCount = 4;
    constructor(
      readonly width: number,
      readonly height: number,
      readonly periodic: boolean
    ) {

    }
    // DX = [-1, 0, 1, 0], DY = [0, 1, 0, -1]
    // dir: 0=left, 1=up, 2=right, 3=down
    neighbor(cell: number, dir: number): number { /* ... */ }
    opposite(dir: number): number { return (dir + 2) % 4; }
  }

  export class GridTopology3D implements Topology {
    readonly cellCount: number;
    readonly directionCount = 6;
    constructor(
      readonly width: number,
      readonly height: number,
      readonly depth: number,
      readonly periodic: boolean
    ) {
      this.cellCount = width * height * depth;
    }
    // DX = [-1, 1, 0, 0, 0, 0]
    // DY = [0, 0, 1, -1, 0, 0]
    // DZ = [0, 0, 0, 0, -1, 1]
    // dir: 0=left, 1=right, 2=up, 3=down, 4=front, 5=back
    neighbor(cell: number, dir: number): number { /* ... */ }
    opposite(dir: number): number {
      // 0↔1, 2↔3, 4↔5
      return dir % 2 === 0 ? dir + 1 : dir - 1;
    }
  }
  ```

  ```typescript
  // src/solver-3d.ts
  import { WFCEngine } from './engine.js';
  import { GridTopology3D } from './topology.js';
  import type { TileRule3D, Heuristic, StepStatus } from './types.js';

  export interface WFCSolver3DOptions {
    width: number;
    height: number;
    depth: number;
    periodic: boolean;
    weights: number[] | Float64Array;
    rules: TileRule3D[];
    heuristic?: Heuristic;
  }

  export class WFCSolver3D {
    private engine: WFCEngine;
    private MX: number;
    private MY: number;
    private MZ: number;

    constructor(options: WFCSolver3DOptions) {
      const { width, height, depth, periodic, weights, rules, heuristic } = options;
      this.MX = width;
      this.MY = height;
      this.MZ = depth;
      const topology = new GridTopology3D(width, height, depth, periodic);
      const { propStart, propLen, propData } = this.buildPropagator(rules, weights.length);
      this.engine = new WFCEngine(topology, new Float64Array(weights), propStart, propLen, propData, heuristic ?? 'mrv');
    }

    private buildPropagator(rules: TileRule3D[], T: number) {
      // 6 directions: left=0, right=1, up=2, down=3, front=4, back=5
      const propagator: number[][][] = [[], [], [], [], [], []];
      for (let t = 0; t < T; t++) {
        const rule = rules.find(r => r.forTile === t);
        propagator[0][t] = rule?.left ?? [];
        propagator[1][t] = rule?.right ?? [];
        propagator[2][t] = rule?.up ?? [];
        propagator[3][t] = rule?.down ?? [];
        propagator[4][t] = rule?.front ?? [];
        propagator[5][t] = rule?.back ?? [];
      }
      // Convert to CSR (same logic as 2D, just D=6)
      // ...
    }

    run(seed: number, limit = -1): boolean { return this.engine.run(seed, limit); }
    stepRun(seed: number, limit = -1): Generator<StepStatus, boolean, unknown> { return this.engine.stepRun(seed, limit); }
    result(): Int32Array { return this.engine.result(); }
    get width() { return this.MX; }
    get height() { return this.MY; }
    get depth() { return this.MZ; }
  }
  ```
